```python
# 避免内存溢出

    writer = csv.writer(file)
    for row in result.all():  # 流式查询
        writer.writerow(...)  # 流式写入
```

### 2. Worker 心跳监控

```python
# 检查 Worker 健康状态
heartbeat_time = datetime.fromisoformat(heartbeat)
age_seconds = (datetime.now(timezone.utc) - heartbeat_time).total_seconds()

if age_seconds < 300:  # 5分钟内
    status = "healthy"
else:
    status = "stale"
```
